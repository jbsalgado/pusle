🚀 Como Usar os Models

1. Estrutura de Pastas

// Estrutura recomendada app/ ├── models/ │ ├── Usuario.php │ ├── Cliente.php │ ├── Produto.php │ ├── Venda.php │ ├── Parcela.php │ ├── Colaborador.php │ ├── CarteiraCobranca.php │ ├── PeriodoCobranca.php │ ├── Comissao.php │ └── ... (todos os outros models) ├── controllers/ ├── views/ └── config/ └── web.php // Configurar autenticação

2. Configuração de Autenticação

O model Usuario implementa IdentityInterface e pode ser usado diretamente para autenticação.
// config/web.php 'components' => [ 'user' => [ 'identityClass' => 'app\models\Usuario', 'enableAutoLogin' => true, ], // ... ],

3. Namespace

⚠️ Importante: Todos os models estão no namespace app\models. Certifique-se de que seus controllers importem corretamente.
namespace app\controllers; use Yii; use app\models\Cliente; use app\models\Produto; use app\models\Venda; // etc...

💡 Exemplos Práticos de Uso

🔐 Exemplo 1: Login de Usuário
// LoginForm.php public function login() { if ($this->validate()) { $usuario = Usuario::findByEmail($this->email); if ($usuario && $usuario->validatePassword($this->senha)) { return Yii::$app->user->login($usuario); } } return false; }
👤 Exemplo 2: Cadastrar Cliente
// ClienteController.php public function actionCreate() { $model = new Cliente(); $model->usuario_id = Yii::$app->user->id; if ($model->load(Yii::$app->request->post()) && $model->save()) { Yii::$app->session->setFlash('success', 'Cliente cadastrado!'); return $this->redirect(['view', 'id' => $model->id]); } return $this->render('create', ['model' => $model]); }
🛒 Exemplo 3: Criar Venda com Itens
// VendaController.php public function actionCreate() { $transaction = Yii::$app->db->beginTransaction(); try { // Criar venda $venda = new Venda(); $venda->usuario_id = Yii::$app->user->id; $venda->cliente_id = $clienteId; $venda->colaborador_vendedor_id = $vendedorId; $venda->valor_total = $valorTotal; $venda->numero_parcelas = 3; $venda->status_venda_codigo = StatusVenda::EM_ABERTO; $venda->save(); // Adicionar itens foreach ($itens as $item) { $vendaItem = new VendaItem(); $vendaItem->venda_id = $venda->id; $vendaItem->produto_id = $item['produto_id']; $vendaItem->quantidade = $item['quantidade']; $vendaItem->preco_unitario_venda = $item['preco']; $vendaItem->save(); } // Gerar parcelas $venda->gerarParcelas(); // Calcular comissão do vendedor Comissao::calcularComissaoVenda($venda->id, $vendedorId); $transaction->commit(); return $this->redirect(['view', 'id' => $venda->id]); } catch (\Exception $e) { $transaction->rollBack(); throw $e; } }
💰 Exemplo 4: Registrar Pagamento de Parcela
// ParcelaController.php public function actionPagar($id) { $parcela = Parcela::findOne($id); if ($parcela->registrarPagamento( $valorPago, $cobradorId, $formaPagamentoId )) { // Registrar no histórico $historico = new HistoricoCobranca(); $historico->parcela_id = $parcela->id; $historico->cobrador_id = $cobradorId; $historico->cliente_id = $parcela->venda->cliente_id; $historico->usuario_id = Yii::$app->user->id; $historico->tipo_acao = HistoricoCobranca::TIPO_PAGAMENTO; $historico->valor_recebido = $valorPago; $historico->save(); // Calcular comissão de cobrança Comissao::calcularComissaoCobranca($parcela->id, $cobradorId); Yii::$app->session->setFlash('success', 'Pagamento registrado!'); } return $this->redirect(['index']); }
📅 Exemplo 5: Distribuir Carteira de Cobrança
// CarteiraController.php public function actionDistribuir() { $periodo = PeriodoCobranca::getPeriodoAtual(); $clientes = Cliente::find() ->where(['usuario_id' => Yii::$app->user->id, 'ativo' => true]) ->all(); foreach ($clientes as $cliente) { $carteira = new CarteiraCobranca(); $carteira->periodo_id = $periodo->id; $carteira->cobrador_id = $cobradorId; // lógica de escolha $carteira->cliente_id = $cliente->id; $carteira->usuario_id = Yii::$app->user->id; $carteira->ativo = true; // Calcular totais do cliente $parcelas = Parcela::find() ->joinWith('venda') ->where(['venda.cliente_id' => $cliente->id]) ->andWhere(['!=', 'status_parcela_codigo', StatusParcela::PAGA]) ->all(); $carteira->total_parcelas = count($parcelas); $carteira->valor_total = array_sum(array_column($parcelas, 'valor_parcela')); $carteira->save(); } Yii::$app->session->setFlash('success', 'Carteiras distribuídas!'); return $this->redirect(['index']); }

✅ Validações Implementadas

✓ Todos os models possuem validações completas
  • Multi-tenant: usuario_id obrigatório em todos
  • E-mails: Validação de formato e unicidade
  • CPF: Apenas números, 11 dígitos, único
  • CEP: Apenas números, 8 dígitos
  • Telefone: Máximo 20 caracteres
  • Valores monetários: Sempre >= 0
  • Percentuais: Entre 0 e 100
  • Cores: Formato hexadecimal (#RRGGBB)
  • Status: Valores restritos (in range)
  • Foreign Keys: exist() em todos os relacionamentos
  • Unicidade: Constraints compostos onde necessário
  • Datas: Formato Y-m-d validado

⚙️ Behaviors Implementados

Models com data_criacao e data_atualizacao usam TimestampBehavior para atualização automática.

Models com timestamps:

🔗 Relacionamentos (Relations)

⚠️ Importante: Todos os relacionamentos usam hasOne() ou hasMany() corretamente. Use with() ou joinWith() para eager loading.

Exemplo de Uso de Relacionamentos

// Eager loading (mais eficiente) $vendas = Venda::find() ->with(['cliente', 'vendedor', 'itens.produto']) ->all(); foreach ($vendas as $venda) { echo $venda->cliente->nome_completo; echo $venda->vendedor->nome_completo; foreach ($venda->itens as $item) { echo $item->produto->nome; } }
